.container {
width: 500px;
overflow-x: auto;
overflow-y: hidden;
white-space: nowrap;
/* Set up points to which scrolling will snap */
-ms-scroll-snap-points-x: snapInterval(0px, 100%);
/* Require that scrolling always end at a snap point */
-ms-scroll-snap-type: mandatory;
}
